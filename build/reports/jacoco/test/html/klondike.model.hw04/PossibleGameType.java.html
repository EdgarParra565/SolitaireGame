package klondike.model.hw04;

/**
 * Enum class represents the two different types of klondike game models.
 * Enum values store a string within that can be used to initialize new model objects.
 */
public enum PossibleGameType {
  BASIC("basic"),
  WHITEHEAD("whitehead");

  private final String value;

  PossibleGameType(String value) {
    this.value = value;
  }

  /**
   * Observer method that returns the game type
   * when given a PossibleGameType via a string.
   *
   * @return a string that will be used to interpret the game model.
   */
  public String getPossibleGameType() {
    return this.value;
  }


}
